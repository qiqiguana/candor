# Summa keyword transliteration
              
# Syntax:
#   "source" => "target"
#   "source".length() > 0 (source cannot be empty.)
#   "target".length() >= 0 (target can be empty.)

# example:
#   "À" => "A"
#   "\u00C0" => "A"
#   "\u00C0" => "\u0041"
#   "ß" => "ss"
#   "\t" => " "
#   "\n" => ""

"_" => " "

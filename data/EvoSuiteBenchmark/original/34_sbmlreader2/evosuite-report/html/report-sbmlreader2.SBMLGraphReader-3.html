139: 			nodeAttributes.setAttribute(species.getId(), "sbml name", species.getName());
140: 			nodeAttributes.setAttribute(species.getId(), "sbml type", "species");

191: 				                                  "reaction-reactant", true);

196: 			List modifiers = reaction.getModifier();

199: 				ModifierSpeciesReference species = (ModifierSpeciesReference) j.next();

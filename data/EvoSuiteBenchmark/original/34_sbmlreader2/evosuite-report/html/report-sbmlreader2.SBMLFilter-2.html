 51:  * SBMLReader extends CyFileFilter for integration into the Cytoscape ImportHandler
 52:  * framework.
 53:  *
 54:  */
 55: public class SBMLFilter extends CyFileFilter {
 56: 	/**
 57: 	 * XGMML Files are Graphs.
 58: 	 */
 59: 	private static String fileNature = ImportHandler.GRAPH_NATURE;
 60: 
 61: 	/**
 62: 	 * File Extensions.
 63: 	 */
 64: 	private static String[] fileExtensions = { "xml", "sbml" };
 65: 
 66: 	/**
 67: 	 * Filter Description.
 68: 	 */
 69: 	private static String description = "SBML files";
 70: 
 71: 	/**
 72: 	 * Constructor.
 73: 	 */
 74: 	public SBMLFilter() {
 75: 		super(fileExtensions, description, fileNature);
 76: 	}
 77: 
 78: 	/**
 79: 	 * Indicates which files the SBMLFilter accepts.
 80: 	 * <p/>
 81: 	 * This method will return true only if:
 82: 	 * <UL>
 83: 	 * <LI>File ends in .xml or .sbml;  and
 84: 	 * <LI>File headers includes the www.sbml.org namespace declaration.
 85: 	 * </UL>
 86: 	 *
 87: 	 * @param file File
 88: 	 * @return true or false.
 89: 	 */
 90: 	public boolean accept(File file) {
 91: 		String fileName = file.getName();
 92: 		boolean firstPass = false;
 93: 
 94: 		//  First test:  file must end with one of the registered file extensions.
 95: 		for (int i = 0; i < fileExtensions.length; i++) {
 96: 			if (fileName.endsWith(fileExtensions[i])) {
 97: 				firstPass = true;
 98: 			}
 99: 		}
100: 
101: 		if (firstPass) {
102: 			//  Second test:  file header must contain the biopax declaration
103: 			try {
104: 				String header = getHeader(file);
105: 
106: 				if (header.indexOf("www.sbml.org") > 0) {
107: 					return true;
108: 				}
109: 			} catch (IOException e) {
110: 			}
111: 		}
112: 
113: 		return false;
114: 	}
115: 
116: 	/**
117: 	 * Gets the appropirate GraphReader object.
118: 	 * If the libsbml is not available use the old sbml library.
119: 	 *
120: 	 * @author Matthias Koenig
121: 	 * @param fileName File Name.
122: 	 * @return GraphReader Object.
123: 	 */
124: 	public GraphReader getReader(String fileName) {
125: 		System.out.println("Getting SBML reader ...");
126: 		try
127: 	    {
128: 		  // Test if libsbml library is available
129: 		  System.loadLibrary("sbmlj");  
130: 		  /* Extra check to be sure we have access to libSBML: */
131: 		  Class.forName("org.sbml.libsbml.libsbml");
132: 	     
133: 	      
134: 	      // Use the libSBML based parser
135: 		  System.out.println("... using 'libsbml parser'");
136: 		  return new SBMLGraphReader2(fileName);
137: 	    }
138: 	    catch (Exception e)
139: 		{ 
140: 	      // If any error occurs use the classic SBML Reader
141: 	      return new SBMLGraphReader(fileName);
142: 	    }
143: 	}
144: }
